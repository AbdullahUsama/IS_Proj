% Section 3: Original Algorithm
\section{Original Algorithm}

\begin{frame}{Original Algorithm: Overview}
    \textbf{Main Steps:}
    \begin{enumerate}
        \item Partition cover image into 8×8 blocks
        \item Store Upper Bound (UB) and Lower Bound (LB) in first block
        \item For each subsequent block:
        \begin{itemize}
            \item Compute median of each column
            \item Calculate mean-of-medians (\textit{Me})
            \item For each pixel pair $(p_1, p_2)$:
            \begin{itemize}
                \item Calculate difference: $D_i = |p_1 - p_2|$
                \item If $D_i \leq Me$: embed data based on MSB pattern
            \end{itemize}
        \end{itemize}
    \end{enumerate}
    
    \vspace{0.5em}

\begin{frame}{MSB-Based Embedding Cases}
    \begin{table}
        \centering
        \small
        \begin{tabular}{ccccl}
            \toprule
            \textbf{Case} & \textbf{MSB($p_1$)} & \textbf{MSB($p_2$)} & \textbf{Bits} & \textbf{Positions} \\
            \midrule
            0 & 0 & 0 & 2 & bit 1 of both pixels \\
            1 & 1 & 0 & 3 & bits 2,3 of $p_1$; bit 1 of $p_2$ \\
            2 & 0 & 1 & 3 & bit 1 of $p_1$; bits 2,3 of $p_2$ \\
            3 & 1 & 1 & 4 & bits 2,3 of both pixels \\
            \bottomrule
        \end{tabular}
    \end{table}
    
    \vspace{1em}
    
    \textbf{Key Insight:}
    \begin{itemize}
        \item Different MSB patterns $\Rightarrow$ different embedding capacities
        \item Adaptive to pixel characteristics
        \item Higher capacity than traditional 1-bit LSB
        \item Maintains imperceptibility by avoiding MSB modification
    \end{itemize}
\end{frame}

\begin{frame}{Mean-of-Medians Threshold}
    \textbf{Purpose:} Adaptive embedding based on local pixel statistics
    
    \vspace{1em}
    
    \textbf{Computation:}
    \begin{enumerate}
        \item For 8×8 block, calculate median of each column (8 medians)
        \item Compute mean of these 8 medians: $Me = \frac{1}{8}\sum_{i=1}^{8} \text{median}_i$
        \item Use $Me$ as threshold for pixel difference
    \end{enumerate}
    
    \vspace{1em}
    
    \textbf{Embedding Criterion:}
    $$D_i = |p_1 - p_2| \leq Me \Rightarrow \text{Embed data}$$
    
    \vspace{0.5em}
    
    \begin{block}{Advantage}
        Only embed in pixel pairs with small differences, maintaining local smoothness and reducing detectability
    \end{block}
\end{frame}

\begin{frame}{Enhancement Strategy}
    \begin{columns}
        \column{0.5\textwidth}
        \textbf{Security Enhancement:}
        \begin{block}{AES-CTR Encryption}
            \begin{itemize}

        \end{block}
        
        \column{0.5\textwidth}
        \textbf{Robustness Enhancement:}
        \begin{block}{Edge-Adaptive Embedding}
            \begin{itemize}
                \item Sobel edge detection
                \item Prioritize high-gradient regions
                \item Embed where changes are imperceptible
                \item Reduces statistical signatures
                \item Better PSNR
            \end{itemize}
        \end{block}
    \end{columns}
\end{frame}

\begin{frame}{Edge-Adaptive Embedding}
    \textbf{Motivation:} Human Visual System (HVS) is less sensitive to changes in textured/edge regions
    
    \vspace{1em}
    
    \textbf{Implementation:}
    \begin{enumerate}
        \item Compute Sobel edge magnitude: $\text{Edge} = \sqrt{(\nabla_x)^2 + (\nabla_y)^2}$
        \item Calculate edge score for each 8×8 block
        \item Sort blocks by edge intensity (descending)
        \item Embed only in blocks with $\text{edge\_score} \geq \text{threshold}$
    \end{enumerate}
    
    \vspace{1em}
    
    \begin{columns}
        \column{0.5\textwidth}
        \textbf{Benefits:}
        \begin{itemize}
            \item Higher PSNR (50-55 dB)
            \item Lower histogram deviation
            \item Reduced RS detection
            \item Better perceptual quality
        \end{itemize}
        
        \column{0.5\textwidth}
        \textbf{Trade-off:}
        \begin{itemize}
            \item Reduced capacity
            \item Edge regions only
            \item Cover image dependent
        \end{itemize}
    \end{columns}
\end{frame}

    \begin{lstlisting}[language=Python, basicstyle=\tiny\ttfamily, keywordstyle=\color{blue}, commentstyle=\color{darkgreen}]
# Edge-adaptive embedding
def encode(self, cover_image, payload_bytes):
    # Compute edge map using Sobel
    edge_map = self._compute_edge_map(cover_image)
    
    # Partition into 8x8 blocks
    blocks = self._partition_into_blocks(cover_image)
    
    # Sort blocks by edge intensity
    blocks_sorted = sorted(blocks, 
                          key=lambda b: edge_score(b), 
                          reverse=True)
    
    # Embed in high-edge blocks only
    for block in blocks_sorted:
        if edge_score(block) >= threshold:
            Me = self._compute_mean_of_medians(block)
            for p1, p2 in pixel_pairs(block):
                if abs(p1 - p2) <= Me:
                    case = self._get_embedding_case(p1, p2)
                    embed_bits(p1, p2, payload_bits, case)
    \end{lstlisting}
